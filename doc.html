Examples

Get browser information

            
              const Joo = require('joo');
              console.log(Joo.get());

              //output
              {
                   "bot": false,
                   "mobile": false,
                   "name": "chrome",
                   "version": "68.0.3440",
                   "versionNumber": 68.0344,
                   "os": "Windows 10"
              }

Check browser 

            
              const Joo = require('joo');
              Joo.is([
                'chrome > 70.0.3440',
                'firefox > 32',
                'ie > 10'
                ], is => {
                  console.log( is ? 'Valid' : 'No Valid');
              });

Watch When any error happend

            
              const Joo = require('joo');
              Joo.error( error => {
                console.log(error);
              });
